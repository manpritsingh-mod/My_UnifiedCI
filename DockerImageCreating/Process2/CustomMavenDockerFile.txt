FROM maven:3.8.6-openjdk-11

# Install additional tools
RUN apt-get update && apt-get install -y \
    git \
    curl \
    && rm -rf /var/lib/apt/lists/*

# Pre-download common Maven plugins (speeds up builds)
RUN mvn dependency:resolve-plugins \
    org.apache.maven.plugins:maven-compiler-plugin:3.11.0 \
    org.apache.maven.plugins:maven-surefire-plugin:3.0.0

WORKDIR /workspace

ARG MAVEN_VERSION=3.8.6
ARG JAVA_VERSION=11
FROM openjdk:${JAVA_VERSION}-jdk-slim

# Install system dependencies
RUN apt-get update && apt-get install -y \
    curl \
    git \
    && rm -rf /var/lib/apt/lists/*

# Install Maven
RUN curl -L https://archive.apache.org/dist/maven/maven-3/${MAVEN_VERSION}/binaries/apache-maven-${MAVEN_VERSION}-bin.tar.gz -o maven.tar.gz \
    && tar -xzf maven.tar.gz \
    && mv apache-maven-${MAVEN_VERSION} /opt/maven \
    && rm maven.tar.gz

# Set environment variables
ENV MAVEN_HOME=/opt/maven
ENV PATH=$PATH:$MAVEN_HOME/bin

# Create workspace directory
WORKDIR /workspace
